DECLARE STR1, STR2 : STRING
DECLARE CHAR1, CHAR2, CHAR3 : CHARACTER
DECLARE I : INTEGER

STR1 <-- ""
STR2 <-- ""
CHAR1 <-- ''
CHAR2 <-- ''
CHAR3 <-- ''
I <-- 0

INPUT "ENTER THE STRING YOU WANT TO PROCESS: "
INPUT "ENTER THE WORD YOU WANT TO CHANGE : "
INPUT "ENTER THE WORD FOR YOUR REPLACEMENT: "


FOR I <-- 1 TO LEN(STR1)
    CHAR1 <-- MID(STR1, I, 1)
   IF CHAR1 <> CHAR2 
 THEN
   STR2 <-- STR2 + CHAR2
 ELSE
   STR2 <-- STR2 + CHAR3
  
  END IF

  NEXT I 


OUTPUT "HERE IS YOUR CHANGED WORD: ", STR2
